# Cross-service version constraints (post-v1.0.0-claw audit fix W2.18).
#
# Use from any service requirements.txt via:
#   -c ../../_constraints.txt
#
# These are pin-floors / pin-ceilings for libraries that are shared across
# multiple Python services. Version skew across services on these libs has
# real consequences:
#   - rdkit / drfp: SMILES canonicalization differs across versions, which
#     produces duplicate :Compound nodes in the KG for the same molecule.
#   - pydantic / fastapi: schema-validation behavior changes between minors.
#   - psycopg: must be psycopg3 across the board (v2 has a different API).
#
# Production deployments should override these via pip install -c <file>
# pointing at a frozen lockfile.
#
# NOTE: pip 26.x removed support for "extras" in constraints files (see
# https://github.com/pypa/pip/issues/8210). The extras (psycopg[binary],
# uvicorn[standard]) are declared on the per-service requirement lines
# instead — the constraint here just provides the version pin.

# Web framework
fastapi>=0.115,<0.120
uvicorn>=0.32,<0.40

# Validation / settings
pydantic>=2.8,<3
pydantic-settings>=2.4,<3

# HTTP
httpx>=0.27,<0.30

# Postgres
psycopg>=3.2,<4

# Chemistry
rdkit>=2024.9
drfp>=0.3.5,<1

# Structured logging (PR-logging). python-json-logger 3.x supports the
# `orjson` serializer for ~2x faster encoding on hot paths (per-projector
# ack, per-tool span). Both are kept in lockstep across services so the
# JSON shape and timestamp format are identical wherever a log line lands.
python-json-logger>=3.2,<4
orjson>=3.10,<4
